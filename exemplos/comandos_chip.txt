# === Item (a): apenas condução pelas pernas ===

# Calcula a condutância de uma perna (auxiliar):
g_cond k=380 A=7.854e-7 L=0.02
# Resultado esperado: G ≈ 0.014923 W/K por perna.
# Como há 12 pernas em paralelo: G_total = 12 × 0.014923 = 0.17908 W/K

# Adiciona o nó do componente (com a fonte Q = 4 W) e o nó da placa:
node Tc diffusion Q=4 T_init=30
node Tp boundary T_fixed=44

# Liga os dois com a condutância de condução das 12 pernas:
link Tc Tp cond G=0.17908

# Verifica a rede e resolve:
show
solve
# Resultado esperado: Tc ≈ 66.34 °C (livro: 66.3 °C)

# === Item (b): acrescenta convecção para o ar ===

# Condutância convectiva: A = (0.030)² = 9e-4 m²
g_conv h=30 A=9e-4
# Resultado esperado: G_conv ≈ 0.027 W/K

# Adiciona o nó de fronteira do ar e a ligação convectiva:
node Tar boundary T_fixed=20
link Tc Tar conv G=0.027

# Resolve novamente com a rede expandida:
solve
# Resultado esperado: Tc ≈ 60.27 °C
